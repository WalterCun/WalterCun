```python
from datetime import datetime

class Bio:
    """📋 Perfil profesional de Walter Cun - Datos clave para reclutadores"""

    def __init__(self, name: str, location: str, role: str, experience: int):
        self.name = name
        self.location = location
        self.role = role
        self.experience = experience  # años desde 2019
    
    def resume(self) -> dict:
        """Retorna información estructurada para filtrado rápido"""
        return {
            "👤 Nombre": f"{self.name}",
            "📍 Ubicación": f"{self.location} (UTC-5) • Remoto global",
            "💼 Rol Actual": "Senior Backend Developer (Freelance)",
            "🎯 Especialización": "Python • Django • APIs • Arquitectura",
            "🚀 Enfoque Actual": "PMS Open-Source • Scraping • Microservicios",
            "🌱 Aprendiendo": "Arquitectura Cloud • MLOps • Inglés técnico",
            "🤝 Busco": "Proyectos Python/TS • Equipos remotos • Impacto real",
            "💡 Valor Agregado": "Código limpio • Docs claras • Visión de producto",
            "📬 Contacto": "waltercunbustamante@gmail.com • GitHub: @WalterCun",
        }

def main():
    """Ejecuta y muestra el perfil formateado"""
    profile = Bio(
        name="Walter Cun Bustamante",
        location="Cuenca, Ecuador", 
        role="Full Stack Developer",
        experience= datetime.now().year - 2019 
    )
    
    print("🔍 PERFIL PARA RECLUTADORES\n" + "="*40)
    for key, value in profile.resume().items():
        print(f"{key}: {value}")

if __name__ == "__main__":
    main()
```
