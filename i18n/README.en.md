from datetime import datetime

class Bio:
    """📋 Walter Cun's Professional Profile - Key Data for Recruiters"""

    def __init__(self, name: str, location: str, role: str, experience: int):
        self.name = name
        self.location = location
        self.role = role

            "🤝 Looking For": "Python/TS Projects • Remote Teams • Real Impact",
            "💡 Added Value": "Clean Code • Clear Docs • Product Vision",
            "📬 Contact": "waltercunbustamante@gmail.com • GitHub: @WalterCun",
        }

def main():
    """Executes and displays the formatted profile"""
    profile = Bio(
        name="Walter Cun Bustamante",
        location="Cuenca, Ecuador", 
        role="Full Stack Developer",
        experience= datetime.now().year - 2019 
    )
    
    print("🔍 PROFILE FOR RECRUITERS\n" + "="*40)
    for key, value in profile.resume().items():
        print(f"{key}: {value}")

if __name__ == "__main__":
    main()
